function some_cond(int $i): bool {
    return $i++ % 2 === 1;
}

final class C {}

function maybe_object(bool $test): ?C {
    return $test ? new C() : null;
}

function main(bool $input, int $x, string $foo): int {
    if ($x !== 0 && $input && maybe_object($input) is nonnull && some_cond($x)) {
        echo "test";
    }

    if ($input || maybe_object($input) is null || !some_cond($x)) {
        echo "test";
    }

    if (maybe_object($input) is nonnull) {
        echo "test";
    }

    if (maybe_object($input) is null) {
        echo "test";
    }

    if ($input && !(maybe_object($input) is nonnull || some_cond($x))) {
        echo "test";
    }

    if ($x === 0) {
        echo "test";
    }

    $bar = 4;
    if (rand(0,1) > 0) {
        $bar = 'baz';
    }

    if ($bar) {
        echo "test";
    }

    if ((int)preg_match('/\s/', $foo) > 0) {
        echo "test";
    }

    if ((int)preg_match('/\s/', $foo) > 0) {
        echo "test";
    }

    if ($input || (int)preg_match('/\s/', $foo) === 0) {
        echo "test";
    }

    return maybe_object($input) is nonnull ? 5 : 4;
}
